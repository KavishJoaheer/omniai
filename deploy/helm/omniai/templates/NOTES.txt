    kubectl port-forward -n {{ .Release.Namespace }} svc/{{ include "omniai.fullname" . }}-frontend 8080:8080
{{- end }}

Verify:
    kubectl exec -it deploy/{{ include "omniai.fullname" . }}-api -- curl -s localhost:9380/v1/health

Scale workers:
    kubectl scale -n {{ .Release.Namespace }} deploy/{{ include "omniai.fullname" . }}-worker --replicas=N

Migrations run automatically on api pod startup. If you need to run them
ad-hoc (e.g. before deploying a new image):
    kubectl run -n {{ .Release.Namespace }} omniai-migrate --rm -i --tty \
        --image={{ .Values.api.image.repository }}:{{ .Values.api.image.tag }} \
        --env="OMNIAI_ROLE=migrate" \
        --restart=Never \
        --command -- /usr/local/bin/entrypoint.sh
